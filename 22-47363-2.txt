#include <GL/glut.h>
#include <math.h>

#define PI 3.14159265

void drawCar() {
    glBegin(GL_POLYGON);
    glColor3f(1.0, 0.0, 0.0);
    glVertex2f(0, 0);
    glColor3f(0.0, 1.0, 0.0);
    glVertex2f(0, 50);
    glColor3f(0.0, 0.0, 1.0);
    glVertex2f(50, 50);
    glColor3f(1.0, 0.0, 1.0);
    glVertex2f(70, 30);
    glColor3f(1.0, 1.0, 0.0);
    glVertex2f(70, 0);
    glEnd();


    float radius = 10;
    int i;
    glColor3f(0.0, 1.0, 0.0);


    glBegin(GL_POLYGON);
    for (i = 0; i < 360; i++) {
        float theta = i * PI / 180;
        glVertex2f(15 + radius * cos(theta), -2 + radius * sin(theta));
    }
    glEnd();

    glBegin(GL_POLYGON);
    for (i = 0; i < 360; i++) {
        float theta = i * PI / 180;
        glVertex2f(55 + radius * cos(theta), -2 + radius * sin(theta));
    }
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    // Drawing the first car (without any transformation)
    glPushMatrix();
    glTranslatef(50, 120, 0);
    drawCar();
    glPopMatrix();

    // Drawing the second car (with rotation and scaling)
    glPushMatrix();
    glTranslatef(50, 100, 0);

    glScalef(3.0, 4.0, 1.0);

    glRotatef(175, 0, 0, 1);

    drawCar();
    glPopMatrix();

    glFlush();
}

void init() {
    glClearColor(0.0, 0.0, 0.0, 1.0); // Black background
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(-200, 200, -300, 300); // Match image's coordinate system
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(800, 600);
    glutCreateWindow("Car and Rotated Car");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
